---
title: "Create Subtask"
api: "POST /api/subtasks"
description: "Create a new subtask within a task"
---

Creates a new subtask associated with a parent task. Subtasks help break down larger tasks into smaller, manageable units.

## Authentication

Requires authentication via session cookie or Bearer token.

## Request Body

<ParamField body="taskId" type="long" required>
  The ID of the parent task
</ParamField>

<ParamField body="title" type="string" required>
  The title of the subtask (max 255 characters)
</ParamField>

<ParamField body="description" type="string">
  Detailed description of the subtask
</ParamField>

<ParamField body="tagIds" type="array">
  List of tag IDs to associate with the subtask
</ParamField>

<ParamField body="durationTagId" type="integer">
  ID of the duration tag indicating estimated time
</ParamField>

<ParamField body="status" type="string" default="pending">
  Initial status of the subtask. Options: `pending`, `in_progress`, `completed`
</ParamField>

<ParamField body="displayOrder" type="integer" default="0">
  Order position for displaying the subtask in lists
</ParamField>

## Response

<ResponseField name="subtaskId" type="long">
  Unique identifier for the created subtask
</ResponseField>

<ResponseField name="taskId" type="long">
  ID of the parent task
</ResponseField>

<ResponseField name="title" type="string">
  The subtask title
</ResponseField>

<ResponseField name="description" type="string">
  The subtask description
</ResponseField>

<ResponseField name="tags" type="array">
  Array of tag objects associated with the subtask
</ResponseField>

<ResponseField name="durationTagId" type="integer">
  ID of the duration tag
</ResponseField>

<ResponseField name="status" type="string">
  Current status of the subtask
</ResponseField>

<ResponseField name="completedAt" type="datetime">
  Timestamp when the subtask was completed (null if not completed)
</ResponseField>

<ResponseField name="displayOrder" type="integer">
  Display order position
</ResponseField>

<ResponseField name="createdAt" type="datetime">
  Timestamp when the subtask was created
</ResponseField>

<ResponseField name="updatedAt" type="datetime">
  Timestamp when the subtask was last updated
</ResponseField>

```bash cURL
  }'
```

```python Python
import requests

url = "https://api.taskpro.com/api/subtasks"
headers = {
    "Authorization": "Bearer YOUR_TOKEN",
    "Content-Type": "application/json"
}
payload = {
    "taskId": 123,
    "title": "Research API options",
    "description": "Look into REST vs GraphQL for the project",
    "status": "pending",
    "durationTagId": 2,
    "tagIds": [5, 8],
    "displayOrder": 1
}

response = requests.post(url, json=payload, headers=headers)
print(response.json())
```
