## POST /api/tasks

Creates a new task for the authenticated user.

### Authentication

Required. Use JWT Bearer token in the Authorization header.

### Request Body

<ParamField body="title" type="string" required>
  The title of the task (max 255 characters)
</ParamField>

<ParamField body="description" type="string">
  Detailed description of the task
</ParamField>

<ParamField body="folderId" type="integer">
  ID of the folder to place the task in
</ParamField>

<ParamField body="tagIds" type="array">
  Array of tag IDs to associate with the task
</ParamField>

<ParamField body="durationTagId" type="integer">
  ID of the duration tag for time estimates
</ParamField>

<ParamField body="priority" type="integer" default="0">
  Priority level of the task (higher numbers = higher priority)
</ParamField>

<ParamField body="status" type="string" default="pending">
  Initial status of the task. Valid values: `pending`, `in_progress`, `completed`, `cancelled`
</ParamField>

<ParamField body="dueDate" type="string">
  Due date in ISO 8601 format (YYYY-MM-DD)
</ParamField>

<ParamField body="dueTime" type="string">
  Due time in ISO 8601 format (HH:MM:SS)
</ParamField>

<ParamField body="displayOrder" type="integer" default="0">
  Order for displaying the task in lists
</ParamField>

### Response

<ResponseField name="taskId" type="integer">
  Unique identifier for the created task
</ResponseField>

<ResponseField name="userId" type="integer">
  ID of the user who owns the task
</ResponseField>

<ResponseField name="folderId" type="integer">
  ID of the folder containing the task
</ResponseField>

<ResponseField name="tags" type="array">
  Array of tag objects associated with the task
</ResponseField>

<ResponseField name="title" type="string">
  The task title
</ResponseField>

<ResponseField name="description" type="string">
  The task description
</ResponseField>

<ResponseField name="durationTagId" type="integer">
  ID of the duration tag
</ResponseField>

<ResponseField name="priority" type="integer">
  Priority level of the task
</ResponseField>

<ResponseField name="status" type="string">
  Current status: `pending`, `in_progress`, `completed`, or `cancelled`
</ResponseField>

<ResponseField name="dueDate" type="string">
  Due date in ISO 8601 format
</ResponseField>

<ResponseField name="dueTime" type="string">
  Due time in ISO 8601 format
</ResponseField>

<ResponseField name="completedAt" type="string">
  Timestamp when the task was completed (null if not completed)
</ResponseField>

<ResponseField name="displayOrder" type="integer">
  Display order in lists
</ResponseField>

<ResponseField name="isArchived" type="boolean">
  Whether the task is archived
</ResponseField>

<ResponseField name="createdAt" type="string">
  Timestamp when the task was created
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Timestamp when the task was last updated
</ResponseField>

### Example Request

```bash
curl -X POST https://api.taskpro.com/api/tasks \
  -H "Authorization: Bearer YOUR_JWT_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "title": "Complete API documentation",
    "description": "Write comprehensive API docs for all endpoints",
    "folderId": 5,
    "tagIds": [1, 3],
    "priority": 2,
    "status": "pending",
    "dueDate": "2026-03-20",
    "dueTime": "17:00:00"
  }'
```
